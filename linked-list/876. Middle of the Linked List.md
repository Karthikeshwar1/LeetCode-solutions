class Solution {
public:
    ListNode* middleNode(ListNode* head) {
        int i = 1;
        ListNode *slow = head;
        ListNode *fast = head;
        
        while (1) {
            if (fast->next == NULL) {
                break;
            }
            if (fast->next->next == NULL) {
                i++;
                break;
            }
            i += 2;
            slow = slow->next;
            fast = fast->next->next;
        }
        
        // cout << i << endl;
        return (i%2 == 0 ? slow->next : slow);
    }
};
