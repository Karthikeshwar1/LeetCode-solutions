class Solution {
public:
    bool areAlmostEqual(string s1, string s2) {
        if (s1 == s2) {
            return true;
        }
        
        if (s1.size() != s2.size()) {
            return false;
        }
        
        for (int i = 0; i < s1.size(); i++) {
            for (int j = i+1; j < s1.size(); j++) {
                string s = s1;
                s[i] = s1[j];
                s[j] = s1[i];
                if (s == s2) {
                    return true;
                }
            }    
        }
        
        return false;
    }
};
