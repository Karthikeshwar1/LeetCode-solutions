class Solution {
public:
    bool checkInclusion(string s1, string s2) {
        multiset<char> ms1, ms2;
        
        for (char c : s1) {
            ms1.insert(c);
        }
        
        for (int i = 0, j = 0; j < s2.size(); ) {
            ms2.insert(s2[j]);
            j++;
            
            if (j - i == s1.size()) {
                i++;
                
                if (ms1 == ms2) {
                    return true;
                }
                
                if (i > 0) {
                    ms2.erase(ms2.find(s2[i-1]));
                }
            }
        }
        
        return false;
    }
};
